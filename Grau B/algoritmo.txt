função somaA(n, placas, i) retorna inteiro
    soma = -1
    se i >= n então
        retorne -1
    senão se a soma da linha de cima de placas for igual à soma da linha de baixo então
        soma = soma da linha de cima
    somaSemVirarPlaca = solve(n, placas, i+1)
    somaVirandoAPlaca = solve(n, placas com os valores na posição i invertidos, i+1)
    retorne o máximo entre soma, somaSemVirarPlaca e somaVirandoAPlaca

função solveRemovendoPlacaI(placas, i) retorna inteiro

placas[1][i].remover
return somaA(len(placas[0]), placas, 0)
